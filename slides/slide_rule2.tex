\begin{frame}{Rule 2: Equity --- Who Should Be Prioritized?}
    \begin{center}
        \textbf{Policy should prioritize groups facing the highest \textbf{delay burden}.}
    \end{center}

    \vspace{2em}

    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{Equity Principle}
            \begin{itemize}
                \item Individuals face unequal behavioral costs of delay (\textbf{MWTA})
                \item Some groups experience higher implicit ``time tax''
                \item Equal treatment can lead to unequal outcomes
            \end{itemize}
        \end{column}
        
        \hfill
        
        \begin{column}{0.48\textwidth}
            \textbf{Policy Implication}
            \begin{itemize}
                \item Prioritize groups with higher \textbf{MWTA}
                \item Target interventions toward high-burden populations
                \item Combine \textbf{efficiency} with distributional objectives
            \end{itemize}
        \end{column}
    \end{columns}
    
    \vfill
    \begin{center}
        \textit{\textbf{Equity} requires targeting those who bear the greatest cost of delay.}
    \end{center}
\end{frame}
