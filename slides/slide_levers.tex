\begin{frame}{Dual Policy Levers: Speed vs. Compensation}
    \begin{center}
        \textbf{Vaccination policy can act on both time and incentives.}
    \end{center}

    \vspace{1em}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{Lever 1: \textbf{Speed-up} (Reduce \textbf{Delay})}
            \begin{itemize}
                \item Reduce waiting time to vaccination
                \item Lower administrative and logistical frictions
                \item Directly increase the value of immediate protection
            \end{itemize}
        \end{column}
        
        \hfill
        
        \begin{column}{0.48\textwidth}
            \textbf{Lever 2: \textbf{Compensation} (\textbf{Incentives})}
            \begin{itemize}
                \item Provide monetary or in-kind \textbf{Incentives}
                \item Compensate individuals for delayed protection
                \item Offset the behavioral cost of waiting (MWTA)
            \end{itemize}
        \end{column}
    \end{columns}
    
    \vfill
    \begin{center}
        \textit{These two levers provide alternative ways to increase vaccine uptake.}
    \end{center}
\end{frame}
